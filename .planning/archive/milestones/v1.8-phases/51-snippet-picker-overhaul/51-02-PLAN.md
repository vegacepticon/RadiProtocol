```typescript
// src/views/snippet-tree-picker.ts
import type { App } from 'obsidian';
import type { SnippetService } from '../snippets/snippet-service';

export type SnippetTreePickerMode = 'folder-only' | 'file-only' | 'both';

export interface SnippetTreePickerResult {
  kind: 'folder' | 'file';
  /** Path relative to options.rootPath (NOT vault-relative).
   *  For folders: empty string when the picker rootPath itself is selected (rare; only in folder-only mode).
   *  For files: includes extension (D-03 of Phase 51 — extension kept in stored path). */
  relativePath: string;
}

export interface SnippetTreePickerOptions {
  app: App;
  snippetService: SnippetService;
  /** Container element where the picker mounts. Picker calls container.empty() on mount. */
  container: HTMLElement;
  /** D-08/D-09 mode discriminator. */
  mode: SnippetTreePickerMode;
  /** Vault-absolute root path the picker is anchored at (e.g. settings.snippetFolderPath
   *  or settings.snippetFolderPath + '/' + node.subfolderPath in Runner file-only mode). */
  rootPath: string;
  /** Optional initial selection for highlighting + initial drill restoration.
   *  Relative to rootPath (same convention as SnippetTreePickerResult.relativePath). */
  initialSelection?: string;
  /** D-08 selection callback. Folder selection emits in folder-only and both modes only;
   *  file selection emits in file-only and both modes only. Drilling into a folder does
   *  NOT call onSelect (drill is local nav per D-05/D-06 Phase 30 semantics). */
  onSelect: (result: SnippetTreePickerResult) => void;
}

export class SnippetTreePicker {
  constructor(options: SnippetTreePickerOptions);

  /** Builds DOM into options.container. Idempotent — re-call to refresh after external state change. */
  async mount(): Promise<void>;

  /** Detaches DOM event listeners and clears container.
   *  Callers MUST call this before discarding the picker (Pattern: editor-panel-view re-renders). */
  unmount(): void;
}
```

    - Folder row click (drill view): push folder name to internal drillPath array, re-render listing of new currentAbs path

    - Breadcrumb shows the rootPath-relative path (or "/" when at rootPath)
    - File row click (file-only / both modes): call `onSelect({ kind: 'file', relativePath: <vault-relative path with rootPath stripped> })`
    - "Select this folder" affordance (folder-only / both modes): when drilled into a folder, a button at the top of the listing commits the current folder via `onSelect({ kind: 'folder', relativePath: <currentDrillPath joined with '/'> })`. Wording: «Выбрать эту папку».

    File-row glyph dispatch (Phase 35 MD-01 preserved):
    - `.md` files render with prefix glyph `📝` (Phase 35 MD-01 marker)
    - `.json` files render with prefix glyph `📄`
    - Any other extension (defensive — should not occur given snippet-service load gate) renders with `📄` (default)
    - Folder rows render with `📁` (unchanged from existing convention)
    - Dispatch is by file basename extension (lowercased — `.MD` matches `.md`); no I/O required (no snippet-service.load call to discover kind — extension is sufficient and avoids N+1 loads)

    Search behaviour:
    - Search input always rendered at top, above breadcrumb
    - Empty query (incl. whitespace-only after trim): show drill view at current drillPath (D-12 — DOES NOT reset to rootPath)
    - Non-empty query: switch to search results view; collect candidates via `snippetService.listFolderDescendants(currentAbsRootPath)` where currentAbsRootPath is `options.rootPath` (NOT the drillPath — search is tree-wide from picker root, not from drill cursor)
    - Filter algorithm (D-10): `entry.toLowerCase().includes(query.trim().toLowerCase())` against the BASENAME segment of each entry path. Folder match = folder basename includes query. File match = file basename (with extension) includes query.
    - Filter by mode (D-09): folder-only hides files; file-only hides folders; both shows both
    - Result row layout (D-11): two-line. Primary = name segment (folder basename or file basename WITH extension — `ct-routine.md`) prefixed by the dispatched glyph.     - Result row click (D-12): file → onSelect; folder → drill into it AND clear the search input (set drillPath to the folder's relative segments and re-mount drill view)
    - User clears search manually (input becomes empty after typing): restore drill view at CURRENT drillPath (D-12)
    - Empty results: show muted "Ничего не найдено" message

    - constructor stores options; does NOT mount automatically
    - mount() can be called multiple times (idempotent — empties container first)
    - unmount() removes DOM and any debounce timers; safe to call when never mounted
    - Drill state + search state are instance-private and reset on each mount() call (D-08: "Component owns its own drill-state and search-state (reset on each open, not persisted across Node Editor re-renders)")
