    ```typescript
    describe('createFolder', () => {
      it('creates the folder via ensureFolderPath inside the root', async () => {
        const { vault, folderSet } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await svc.createFolder('.radiprotocol/snippets/new-folder');
        expect(folderSet.has('.radiprotocol/snippets/new-folder')).toBe(true);
      });
      it('is idempotent — a second call does not throw', async () => {
        const { vault } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await svc.createFolder('.radiprotocol/snippets/x');
        await expect(svc.createFolder('.radiprotocol/snippets/x')).resolves.toBeUndefined();
      });
      it('rejects a path outside the root', async () => {
        const { vault } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await expect(svc.createFolder('../escape')).rejects.toThrow(/createFolder rejected/);
      });
    });

    describe('deleteFolder', () => {
      it('trashes the folder via vault.trash(folder, false)', async () => {
        const { vault } = makeVault({ folders: ['.radiprotocol/snippets/sub'] });
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await svc.deleteFolder('.radiprotocol/snippets/sub');
        expect(vault.trash).toHaveBeenCalledTimes(1);
        expect((vault.trash as any).mock.calls[0][1]).toBe(false);
      });
      it('is a no-op for unsafe paths', async () => {
        const { vault } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await svc.deleteFolder('../escape');
        expect(vault.trash).not.toHaveBeenCalled();
      });
      it('is a no-op when the folder does not exist', async () => {
        const { vault } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        await svc.deleteFolder('.radiprotocol/snippets/missing');
        expect(vault.trash).not.toHaveBeenCalled();
      });
    });

    describe('listFolderDescendants', () => {
      it('returns files, folders, and total', async () => {
        const { vault } = makeVault({
          files: { '.radiprotocol/snippets/a/one.json': '{}', '.radiprotocol/snippets/a/b/two.md': '' },
          folders: ['.radiprotocol/snippets/a', '.radiprotocol/snippets/a/b'],
        });
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        const result = await svc.listFolderDescendants('.radiprotocol/snippets/a');
        expect(result.total).toBe(result.files.length + result.folders.length);
        expect(result.files).toEqual(expect.arrayContaining([expect.stringMatching(/one\.json$/), expect.stringMatching(/two\.md$/)]));
      });
      it('returns empty for unsafe path', async () => {
        const { vault } = makeVault();
        const svc = new SnippetService({ vault } as any, { snippetFolderPath: '.radiprotocol/snippets' });
        const result = await svc.listFolderDescendants('../escape');
        expect(result).toEqual({ files: [], folders: [], total: 0 });
      });
    });
    ```
