export interface EdgeLabelDiff {
  edgeId: string;
  currentLabel: string | undefined;
  targetLabel: string | undefined;
}

export interface ReconcileResult {
  diffs: EdgeLabelDiff[];
  newDisplayLabelByAnswerId: Map<string, string | undefined>;
}

export function reconcileEdgeLabels(graph: ProtocolGraph): ReconcileResult;

**Алгоритм (D-04/D-07/D-08/D-09):**

Для каждого `AnswerNode` в `graph.nodes`:

1. Собрать `incomingEdges = graph.edges.filter(e => e.toNodeId === answerId)`. Сохранять порядок из `graph.edges` (детерминированно; `reverseAdjacency` даёт set of sources, но порядок проверки — порядок в `edges`).
2. Найти `pickedLabel` — первую trimmed `edge.label` где `isLabeledEdge(edge) === true` (Phase 49 D-05 re-used). Если ни одного labeled входящего — `pickedLabel = undefined` (D-08 clearing).
3. `currentAnswerLabel = (answer.displayLabel?.trim() || '') || undefined` (trimmed; пустая строка → undefined, симметрично parser-нормализации).
4. Если `pickedLabel` vs `currentAnswerLabel` различаются (`pickedLabel !== currentAnswerLabel`): `newDisplayLabelByAnswerId.set(answerId, pickedLabel)`.
5. Для каждого `e` в `incomingEdges`:
   - `currentEdgeLabel = e.label?.trim() ? e.label.trim() : undefined` (whitespace ≡ unlabeled, Phase 49 D-05).
   - Если `currentEdgeLabel !== pickedLabel`: `diffs.push({ edgeId: e.id, currentLabel: e.label, targetLabel: pickedLabel })`. (`currentLabel` возвращается raw — не trimmed — чтобы caller мог отличить "было `'  '` whitespace" от "было `'Y'`"; `targetLabel` — всегда trimmed или undefined.)
6. Вернуть `{ diffs, newDisplayLabelByAnswerId }`.

**D-07 idempotency contract:** если для КАЖДОГО Answer `pickedLabel === currentAnswerLabel` И все входящие рёбра уже совпадают (trimmed) с `pickedLabel`, то `diffs.length === 0 && newDisplayLabelByAnswerId.size === 0`. Caller, увидев это, НЕ пишет на диск → `modify`-event loop самозатухает после первой реентеративной итерации.

**Почему такой порядок в Map-итерации безопасен:** `graph.nodes` — это Map; iterate через `graph.nodes.values()` с `node.kind === 'answer'` фильтром. Порядок вставки Map сохраняется → детерминированно между запусками.

export interface EdgeLabelDiff {

  currentLabel: string | undefined;
  targetLabel: string | undefined;
}

export interface ReconcileResult {
  diffs: EdgeLabelDiff[];
  newDisplayLabelByAnswerId: Map<string, string | undefined>;
}

/**
 * D-04 edge-wins bi-directional reconcile.
 *
 * For each Answer node:
 *   pickedLabel =
 *     first non-empty (trimmed) incoming edge label
 *     ?? (answer.displayLabel?.trim() || undefined)
 *     ?? undefined
 *
 * Then:
 *   - If pickedLabel differs from answer.displayLabel (trimmed), record
 *     newDisplayLabelByAnswerId.set(answerId, pickedLabel).
 *   - For each incoming edge whose trimmed label !== pickedLabel, push an
 *     EdgeLabelDiff so the caller can re-sync to the picked value.
 *
 * D-07: if no Answer needs a displayLabel change AND no edge diverges from its
 * picked target, the returned result is structurally empty — the caller MUST
 * short-circuit (no disk write), which is how the vault.on('modify') loop
 * self-terminates after the plugin's own write lands (CONTEXT.md D-07).
 *
 * D-18: pure — no vault, no Obsidian import, no side effects.
 */
export function reconcileEdgeLabels(graph: ProtocolGraph): ReconcileResult {
  const diffs: EdgeLabelDiff[] = [];
  const newDisplayLabelByAnswerId = new Map<string, string | undefined>();

  for (const node of graph.nodes.values()) {
    if (node.kind !== 'answer') continue;
    const answer = node as AnswerNode;

    // Deterministic incoming enumeration: filter graph.edges in-array order.
    const incomingEdges: RPEdge[] = graph.edges.filter(e => e.toNodeId === answer.id);

    // D-04 step 1: first non-empty incoming label (whitespace ≡ unlabeled, D-05 reuse)
    const firstLabeled = incomingEdges.find(e => isLabeledEdge(e));
    const edgePick = firstLabeled?.label?.trim() || undefined;

    // D-04 step 2: fallback to trimmed displayLabel if no incoming edge is labeled
    const displayTrim = answer.displayLabel?.trim() || undefined;
    const pickedLabel: string | undefined = edgePick ?? displayTrim;

    // D-04 step 3: propagate displayLabel if it diverges from pickedLabel
    if (displayTrim !== pickedLabel) {
      newDisplayLabelByAnswerId.set(answer.id, pickedLabel);
    }

    // D-04 step 4: push a diff for every incoming edge whose trimmed label !== pickedLabel
    for (const e of incomingEdges) {
      const currentTrim = e.label?.trim() || undefined;
      if (currentTrim !== pickedLabel) {
        diffs.push({
          edgeId: e.id,
          currentLabel: e.label,
          targetLabel: pickedLabel,
        });
      }
    }
  }

  return { diffs, newDisplayLabelByAnswerId };
}
