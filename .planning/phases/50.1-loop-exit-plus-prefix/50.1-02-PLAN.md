    for (const [id, node] of graph.nodes) {

      const exitEdges = outgoing.filter(e => isExitEdge(e));
      const legacyLabeledEdges = outgoing.filter(e => !isExitEdge(e) && isLabeledEdge(e));
      const bodyEdges = outgoing.filter(e => !isExitEdge(e));
