   private applyCanvasPatch(detail: CanvasChangedForNodeDetail): void {
     if (detail.filePath !== this.currentFilePath) return;
     if (detail.nodeId !== this.currentNodeId) return;

     queueMicrotask(() => {
       // Re-check after defer (Phase 42 WR-01 — currentFilePath/currentNodeId may have changed).
       if (detail.filePath !== this.currentFilePath) return;
       if (detail.nodeId !== this.currentNodeId) return;

       if (detail.changeKind === 'deleted') {
         // D-10 — node removed from canvas; return to idle.
         this.pendingEdits = {};
         this.currentNodeId = null;
         this.currentFilePath = null;
         this.formFieldRefs.clear();
         this.pendingCanvasUpdate.clear();
         this.renderIdle();
         return;
       }
       if (detail.changeKind === 'nodeType') {
         // D-09 — kind changed; full re-render via renderNodeForm path.
         const fp = this.currentFilePath;
         const nid = this.currentNodeId;
         this.pendingEdits = {};
         this.formFieldRefs.clear();
         this.pendingCanvasUpdate.clear();
         if (fp !== null && nid !== null) void this.renderNodeForm(fp, nid);
         return;
       }
       // changeKind === 'fields' — D-08 patch each field individually.
       for (const [key, value] of Object.entries(detail.fieldUpdates ?? {})) {
         const el = this.formFieldRefs.get(key);
         if (!el || !el.isConnected) continue;       // Pitfall 2 — defensive against stale refs
         const focused = el.ownerDocument?.activeElement === el;
         if (focused) {
           // D-05 / D-07 — stash for blur-driven flush; D-06 field-level lock applies.
           this.pendingCanvasUpdate.set(key, value);

         this.patchTextareaValue(el, value ?? '');
       }

     el: HTMLInputElement | HTMLTextAreaElement,
     value: string,

     // so the textarea visually matches the new content height without going through

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `EdgeLabelSyncService.subscribe(...)` callback → `EditorPanelView.applyCanvasPatch` | Untrusted (in the integrity sense) external event payload arrives into the view's DOM-mutation surface. |
| Canvas-driven `applyCanvasPatch` write → `pendingEdits` autosave | Two independent write loops (canvas inbound, form outbound) sharing the same DOM. Synchronisation must be invariant-driven, not lock-based. |
| `formFieldRefs` Map references → live HTMLElements that may be detached on re-render | Stale references can survive a renderForm pass. |
| Per-field `blur` handler closures → `this` view instance | Closures live as long as the registered DOM event subscription; if the view is unmounted without `Component.register` cleanup, handlers point at a dead view. |
