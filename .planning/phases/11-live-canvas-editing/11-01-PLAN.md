    ```typescript
      x: number;
      y: number;
      width: number;
      height: number;
      type: string;
      color?: string;

      /** Path of the canvas file currently displayed by this view. */
      file?: { path: string };

       * Presence is checked at runtime via Array.isArray(view.data) before use (D-01).

      /**
       * Triggers a debounced file write of the canvas state.
       * Fire-and-forget — does not return a Promise.
       */
      requestSave(): void;
    }
    ```

    ```typescript
    import type { App } from 'obsidian';
    import type { CanvasViewInternal, CanvasNodeData } from '../types/canvas-internal';

    const PROTECTED_FIELDS = new Set(['id', 'x', 'y', 'width', 'height', 'type', 'color']);

    export class CanvasLiveEditor {
      private readonly app: App;
      private readonly debounceTimers = new Map<string, ReturnType<typeof setTimeout>>();

      constructor(app: App) { ... }

      private getCanvasView(filePath: string): CanvasViewInternal | undefined { ... }

      isLiveAvailable(filePath: string): boolean { ... }

      async saveLive(filePath: string, nodeId: string, edits: Record<string, unknown>): Promise<boolean> { ... }

      private debouncedRequestSave(filePath: string, view: CanvasViewInternal): void { ... }

      destroy(): void { ... }
    }
    ```

    **getCanvasView() implementation (D-01 probe):**
    - Call `this.app.workspace.getLeavesOfType('canvas')`
    - Use `.find()` (not `.some()`) to retain the leaf reference
    - Cast: `const view = leaf?.view as unknown as CanvasViewInternal | undefined`
    - Return `undefined` if view is falsy, OR `!Array.isArray(view.data)`, OR `typeof view.requestSave !== 'function'`
    - Return `view` if all checks pass

    **saveLive() implementation (D-02 + D-03):**
    1. Call `getCanvasView(filePath)` — if undefined, return false (caller uses Strategy A)
    2. Find node: `const node = view.data.find((n: CanvasNodeData) => n.id === nodeId)`
    3. If node is undefined, return false
    4. Check for un-mark path: `const isUnmarking = 'radiprotocol_nodeType' in edits && (edits['radiprotocol_nodeType'] === '' || edits['radiprotocol_nodeType'] === undefined)`
    5. Snapshot keys that will be mutated for rollback:
       ```typescript
       const snapshot: Record<string, unknown> = {};
       // snapshot current values of all keys in edits (or 'existed before' marker)
       ```
    6. Apply edits to node:
       - If `isUnmarking`: delete ALL keys from node that start with `'radiprotocol_'`
       - Otherwise: for each [key, value] in edits — skip if PROTECTED_FIELDS.has(key); if value === undefined delete node[key]; else node[key] = value
    7. Try block: call `this.debouncedRequestSave(filePath, view)`, return true
    8. Catch block: rollback all snapshot keys, throw the error (EditorPanelView will show Notice per D-03)

    **Snapshot strategy for rollback (D-03):**
    Build the snapshot BEFORE mutating:
    ```typescript
    const snapshot = new Map<string, unknown>();
    if (isUnmarking) {
      // Snapshot all radiprotocol_* keys that exist on the node
      for (const key of Object.keys(node)) {
        if (key.startsWith('radiprotocol_')) {
          snapshot.set(key, node[key]);
        }
      }
    } else {
      for (const key of Object.keys(edits)) {
        if (!PROTECTED_FIELDS.has(key)) {
          snapshot.set(key, key in node ? node[key] : '__RP_KEY_NOT_PRESENT__');
        }
      }
    }
    ```
    On rollback:
    ```typescript
    for (const [key, value] of snapshot.entries()) {
      if (value === '__RP_KEY_NOT_PRESENT__') {
        delete node[key];
      } else {
        node[key] = value;
      }
    }
    ```

    **debouncedRequestSave() (D-06):**
    - Clear any existing timer for filePath from debounceTimers Map
    - Set new 500ms setTimeout that calls view.requestSave() then deletes the timer entry
    - Store new timer in debounceTimers Map under filePath key

    **destroy():**
    - Iterate `this.debounceTimers.entries()` and call `clearTimeout()` on each
    - Then clear the Map: `this.debounceTimers.clear()`

    **ESLint compliance:**
    - No `as any` casts — use `as unknown as CanvasViewInternal`
    - No `console.log` — the module is silent; errors are surfaced by callers via Notice
    - Export only `CanvasLiveEditor` class (no default export)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| canvas-live-editor → Obsidian internal Canvas API | Undocumented internal API accessed via runtime cast; untrusted shape |
| edits payload → live node mutation | Caller-supplied key/value pairs applied to live in-memory canvas node |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Tampering | `saveLive()` edits applied to CanvasNodeData | mitigate | PROTECTED_FIELDS set (`id`, `x`, `y`, `width`, `height`, `type`, `color`) checked before every write; keys in this set are silently skipped, never mutated |
| T-11-02 | Tampering | Un-mark path deletes radiprotocol_* keys | accept | Deletion is intentional and user-initiated; no external trigger; scoped to keys starting with `radiprotocol_` only |
| T-11-03 | Denial of Service | Rapid-fire requestSave() calls trigger Obsidian dirty cycle | mitigate | 500ms debounce via Map<string, ReturnType<typeof setTimeout>>; at most one requestSave() per 500ms per canvas file |
| T-11-04 | Tampering | Timer leak after plugin unload triggers requestSave on dead view | mitigate | destroy() clears all timers; called from RadiProtocolPlugin.onunload() in Plan 02 |
| T-11-05 | Information Disclosure | Internal API cast exposes undocumented Obsidian internals | accept | This is a local plugin; no network transmission; undocumented API is read-only detected at runtime with fallback |
</threat_model>
