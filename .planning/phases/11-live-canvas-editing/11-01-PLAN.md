    ```typescript
    // src/types/canvas-internal.d.ts
    // Ambient declarations for Obsidian's undocumented internal Canvas View API.
    // Shape verified via community plugin research (obsidian-advanced-canvas Canvas.d.ts).
    // WARNING: These are internal APIs subject to change. CanvasLiveEditor probes for
    // existence at runtime before use — if absent, the live path silently falls back to Strategy A.

    export interface CanvasNodeData {
      id: string;
      x: number;
      y: number;
      width: number;
      height: number;
      type: string;
      color?: string;
      [key: string]: unknown; // radiprotocol_* fields and any other custom properties
    }

    export interface CanvasData {
      nodes: CanvasNodeData[];
      edges: unknown[];
    }

    /** The Canvas sub-object exposed on CanvasViewInternal as `view.canvas`. */
    export interface CanvasInternal {
      /**
       * Returns a deep copy of the current canvas data (nodes + edges).
       * Edits must be applied to the returned copy and committed back via setData().
       */
      getData(): CanvasData;
      /**
       * Writes the supplied data back to the canvas in-memory state.
       * Must be followed by requestSave() to persist to disk.
       */
      setData(data: CanvasData): void;
      /**
       * Triggers a debounced file write of the canvas state.
       * Fire-and-forget — does not return a Promise.
       */
      requestSave(): void;
    }

    export interface CanvasViewInternal {
      /** Path of the canvas file currently displayed by this view. */
      file?: { path: string };
      /**
       * The Canvas sub-object. Probe for existence via:
       *   typeof leaf.view.canvas?.getData === 'function'
       * This is the stable Pattern B API used by obsidian-advanced-canvas,
       * Obsidian-Link-Nodes-In-Canvas, and enchanted-canvas.
       */
      canvas: CanvasInternal;
    }
    ```

    Notes:

    Implement using Pattern B (`view.canvas.getData()` / `view.canvas.setData()`), with these specific requirements:

    **Class structure:**
    ```typescript
    import type { App } from 'obsidian';
    import type { CanvasViewInternal, CanvasNodeData, CanvasData } from '../types/canvas-internal';

    const PROTECTED_FIELDS = new Set(['id', 'x', 'y', 'width', 'height', 'type', 'color']);

    export class CanvasLiveEditor {
      private readonly app: App;
      private readonly debounceTimers = new Map<string, ReturnType<typeof setTimeout>>();

      constructor(app: App) { ... }

      private getCanvasView(filePath: string): CanvasViewInternal | undefined { ... }

      isLiveAvailable(filePath: string): boolean { ... }

      async saveLive(filePath: string, nodeId: string, edits: Record<string, unknown>): Promise<boolean> { ... }

      private debouncedRequestSave(filePath: string, view: CanvasViewInternal): void { ... }

      destroy(): void { ... }
    }
    ```

    **getCanvasView() implementation (D-01 probe — Pattern B):**
    - Call `this.app.workspace.getLeavesOfType('canvas')`
    - Use `.find()` to retain the leaf reference; cast: `const view = leaf?.view as unknown as CanvasViewInternal | undefined`
    - Return `undefined` if view is falsy OR `typeof view.canvas?.getData !== 'function'`
    - Return `view` if check passes — the `canvas.getData` function presence is sufficient to confirm Pattern B API availability

    **saveLive() implementation (D-02 + D-03 — Pattern B):**
    1. Call `getCanvasView(filePath)` — if undefined, return false (caller uses Strategy A)
    2. Get data copy: `const originalData = view.canvas.getData()` — save for rollback
    3. Get a second copy to mutate: `const updatedData: CanvasData = view.canvas.getData()`
    4. Find node: `const node = updatedData.nodes.find((n: CanvasNodeData) => n.id === nodeId)`
    5. If node is undefined, return false
    6. Check for un-mark path: `const isUnmarking = 'radiprotocol_nodeType' in edits && (edits['radiprotocol_nodeType'] === '' || edits['radiprotocol_nodeType'] === undefined)`
    7. Apply edits to node in the COPY:
       - If `isUnmarking`: delete ALL keys from node that start with `'radiprotocol_'`
       - Otherwise: for each [key, value] in edits — skip if PROTECTED_FIELDS.has(key); if value === undefined delete node[key]; else node[key] = value
    8. Try block: call `view.canvas.setData(updatedData)`, then `this.debouncedRequestSave(filePath, view)`, return true
    9. Catch block: rollback by calling `view.canvas.setData(originalData)`, then re-throw (EditorPanelView shows Notice per D-03)

    **Why two getData() calls:** The first (`originalData`) is kept pristine for rollback. The second (`updatedData`) is the copy we mutate and commit. Both calls happen before any write, so rollback is simply `setData(originalData)` — no key-by-key restoration needed.

    **debouncedRequestSave() (D-06):**
    - Clear any existing timer for filePath from debounceTimers Map
    - Set new 500ms setTimeout that calls `view.canvas.requestSave()` then deletes the timer entry
    - Store new timer in debounceTimers Map under filePath key

    **destroy():**
    - Iterate `this.debounceTimers.entries()` and call `clearTimeout()` on each
    - Then clear the Map: `this.debounceTimers.clear()`

    **ESLint compliance:**
    - No `as any` casts — use `as unknown as CanvasViewInternal`
    - No `console.log` — the module is silent; errors are surfaced by callers via Notice
    - Export only `CanvasLiveEditor` class (no default export)

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Tampering | `saveLive()` edits applied to CanvasNodeData | mitigate | PROTECTED_FIELDS set (`id`, `x`, `y`, `width`, `height`, `type`, `color`) checked before every write; keys in this set are silently skipped, never mutated |
| T-11-02 | Tampering | Un-mark path deletes radiprotocol_* keys | accept | Deletion is intentional and user-initiated; no external trigger; scoped to keys starting with `radiprotocol_` only |
| T-11-03 | Denial of Service | Rapid-fire requestSave() calls trigger Obsidian dirty cycle | mitigate | 500ms debounce via Map<string, ReturnType<typeof setTimeout>>; at most one requestSave() per 500ms per canvas file |
| T-11-04 | Tampering | Timer leak after plugin unload triggers requestSave on dead view | mitigate | destroy() clears all timers; called from RadiProtocolPlugin.onunload() in Plan 02 |
| T-11-05 | Information Disclosure | Internal API cast exposes undocumented Obsidian internals | accept | This is a local plugin; no network transmission; undocumented API is read-only detected at runtime with fallback |
</threat_model>
